CREATE TABLE inbox.meta (
	id bigserial NOT NULL,
	ts timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
	uri uuid NOT NULL,

	fsrarid int8 NOT NULL,
	dsign varchar(512) NOT NULL,
	ipaddress inet NOT NULL,
	CONSTRAINT meta_pkey PRIMARY KEY (id, ts)
);
CREATE UNIQUE INDEX inbox_meta_uri ON inbox.meta USING btree (uri);
